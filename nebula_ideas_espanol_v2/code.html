def execute_llm_chain(
    self,
    prompt: str,
    context: Dict[str, Any]
) -> ChainResult:
    """
    Executes a secure, monitored LLM chain with
    automatic fallback and rate limiting.
    """
    try:
        sanitized_context = self.pii_scrubber.clean(context)
        
        response = self.primary_model.generate(
            prompt, 
            context=sanitized_context
        )
        
        self.telemetry.record_latency(response.timing)
        return response
        
    except RateLimitError:
        logger.warning("Primary model rate limited. Fallback initiated.")
        return self.fallback_model.generate(prompt)
